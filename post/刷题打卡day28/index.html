class Solution {
    public List<Integer> partitionLabels(String s) {
        int[] hash = new int[26];
        List<Integer> res = new ArrayList<>();
        char[] str = s.toCharArray();
        for (char c : str) {
            hash[c - 'a']++;
        }
        int left = 0, right = 0;
        int start = 0;
        while (right < str.length) {
            if (left == right) {
                if (left > 0) {
                    res.add(left - start);
                    start = left;
                }
            }
            char c = str[left];
            if (hash[c - 'a'] > 0) {
                hash[str[right] - 'a']--;
                right++;
            } else {
                left++;
            }
        }
        res.add(str.length - start);
        return res;
    }
}

class Solution {
    public int candy(int[] ratings) {
        int n = ratings.length;
        int[] cand = new int[n];
        int res = 0;
        Arrays.fill(cand, 1);
        for (int i = 1; i < n; i++) {
            if (ratings[i] > ratings[i-1]) {
                cand[i] = cand[i-1] + 1;
            }
        }
        for (int i = n-2; i >= 0; i--) {
            if (ratings[i] > ratings[i+1] && cand[i] <= cand[i+1]) {
                cand[i] = cand[i+1] + 1;
            }
        }
        for (int x : cand) {
            res += x;
        }
        return res;
    }
}

class Solution {
    public int candy(int[] ratings) {
        int[] left = new int[ratings.length];
        int[] right = new int[ratings.length];
        Arrays.fill(left, 1);
        Arrays.fill(right, 1);
        for(int i = 1; i < ratings.length; i++)
            if(ratings[i] > ratings[i - 1]) left[i] = left[i - 1] + 1;
        int count = left[ratings.length - 1];
        for(int i = ratings.length - 2; i >= 0; i--) {
            if(ratings[i] > ratings[i + 1]) right[i] = right[i + 1] + 1;
            count += Math.max(left[i], right[i]);
        }
        return count;
    }
}

class Solution {
    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> {
            if (a[1] < b[1])
                return -1;
            return 1;
        });
        int res = 1;
        int right = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > right) {
                right = points[i][1];
                res++;
            }
        }
        return res;
    }
}

class Solution {
    public int minCameraCover(TreeNode root) {
        // 三个状态
        int[] res = minCam(root);
        return Math.min(res[0], res[2]);
    }

    public int[] minCam(TreeNode root) {
        if (root == null) {
            return new int[] {1000, 0, 0};
        }
        int[] left = minCam(root.left);
        int[] right = minCam(root.right);
        int[] ret = new int[3];
        ret[0] = Math.min(left[1] + right[1], Math.min(left[0] + right[1], left[1] + right[0])) + 1;
        ret[1] = Math.min(left[0] + right[0], left[0] + right[2]);
        ret[1] = Math.min(ret[1], Math.min(left[2] + right[0], left[2] + right[2]));
        ret[2] = Math.min(left[0] + right[0], Math.min(left[0] + right[2], left[2] + right[0]));
        return ret;
    }
}

class Solution {
    private int res;

    public int minCameraCover(TreeNode root) {
        res = 0;
        if (traversal(root) == 0) {
            res++;
        }
        return res;
    }

    public int traversal(TreeNode root) {
        // 定义空节点的状态
        if (root == null)
            return 2;
        int left = traversal(root.left);
        int right = traversal(root.right);
        // 情况1
        if (left == 2 && right == 2)
            return 0;
        // 情况2
        if (left == 0 || right == 0) {
            res++;
            return 1;
        }
        // 情况3
        if (left == 1 || right == 1)
            return 2;
        return -1;
    }
}

class Solution {

    private List<String> res = new ArrayList<>();
    public List<String> restoreIpAddresses(String s) {
        String path = "";
        backtracking(s, 0, 0, path);
        return res;
    }

    public void backtracking(String s, int len, int idx, String path) {
        if (idx == s.length() && len == 4) {
            res.add(path.substring(0, path.length()-1));
        }
        
        if (s.length() - idx < 4 - len || s.length() - idx > (4 - len) * 3)
            return;
        for (int i = 1; i <= 3; i++) {
            if (idx + i <= s.length() && isValid(s, idx, idx+i)) {
                backtracking(s, len+1, idx+i, path + s.substring(idx, idx+i)+".");
            }
        }
    }

    public boolean isValid(String s, int start, int end) {
        if (end - start > 1 && s.charAt(start) == '0')
            return false;
        int num = 0;
        while (start < end) {
            if (s.charAt(start) > '9' || s.charAt(start) < '0')
                return false;
            num = num * 10 + (s.charAt(start) - '0');
            if (num > 255) {
                return false;
            }
            start++;
        }
        return true;
    }
}

import java.util.HashSet;
import java.util.Set;

public class Solution {

    public int firstMissingPositive(int[] nums) {
        int len = nums.length;

        Set<Integer> hashSet = new HashSet<>();
        for (int num : nums) {
            hashSet.add(num);
        }

        for (int i = 1; i <= len ; i++) {
            if (!hashSet.contains(i)){
                return i;
            }
        }

        return len + 1;
    }
}

public class Solution {

    public int firstMissingPositive(int[] nums) {
        int len = nums.length;

        for (int i = 0; i < len; i++) {
            while (nums[i] > 0 && nums[i] <= len && nums[nums[i] - 1] != nums[i]) {
                // 满足在指定范围内、并且没有放在正确的位置上，才交换
                // 例如：数值 3 应该放在索引 2 的位置上
                swap(nums, nums[i] - 1, i);
            }
        }

        // [1, -1, 3, 4]
        for (int i = 0; i < len; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        // 都正确则返回数组长度 + 1
        return len + 1;
    }

    private void swap(int[] nums, int index1, int index2) {
        int temp = nums[index1];
        nums[index1] = nums[index2];
        nums[index2] = temp;
    }
}
