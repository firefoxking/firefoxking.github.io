class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return targetSum == root.val;
        boolean ret = false;
        if (root.left != null)
            ret |= hasPathSum(root.left, targetSum - root.val);
        if (root.right != null)
            ret |= hasPathSum(root.right, targetSum - root.val);
        return ret;
    }
}

class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        if(root==null)return false;
        Deque<TreeNode> stack1 = new LinkedList<>();
        Deque<Integer> stack2 = new LinkedList<>();
        stack1.push(root);
        stack2.push(root.val);
        while(!stack1.isEmpty()){
            int size = stack1.size();
            for(int i=0;i<size;i++){
                TreeNode node = stack1.pop();
                int sum=stack2.pop();
                if(node.left==null && node.right==null && sum == targetSum)
                    return true;
                if(node.right!=null){
                    stack1.push(node.right);
                    stack2.push(sum+node.right.val);
                }
                if(node.left!=null){
                    stack1.push(node.left);
                    stack2.push(sum+node.left.val);
                }
            }
        }
        return false;
    }
}

class Solution {

    private List<List<Integer>> res = new ArrayList<>();
    private int target;

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null)
            return res;
        target = targetSum;
        List<Integer> path = new ArrayList<>();
        backtracking(root, 0, path);
        return res;
    }

    public void backtracking(TreeNode root, int count, List<Integer> path) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && count + root.val == target) {
            path.add(root.val);
            res.add(new ArrayList<>(path));
            path.remove(path.size() - 1);
        }
        path.add(root.val);
        backtracking(root.left, count + root.val, path);
        backtracking(root.right, count + root.val, path);
        path.remove(path.size() - 1);
    }
}

class Solution {
    public int kthSmallest(TreeNode root, int k) {
        TreeNode cur = root;
        Deque<TreeNode> st = new LinkedList<>();
        while (cur != null || !st.isEmpty()) {
            if (cur != null) {
                st.push(cur);
                cur = cur.left;
            } else {
                cur = st.pop();
                if (--k == 0)
                    return cur.val;
                cur = cur.right;
            }
        }
        return -1;
    }
}

968. 监控二叉树

没有想到三个状态定义，一直想着更打家劫舍的定义类似，但打家劫舍只需要定义两个状态就可以解。但我画图尝试的时候也发现这题定义两个状态走不下去。

对于一个节点 root 来说，三个状态定义如下：

  • 状态1：root 安装摄像头
  • 状态2：不安装摄像头，但是被父节点监控
  • 状态3：不安装摄像头，需要被子节点监控

边界条件：

当root == null 时，状态1 为MAX_VALUE，状态2，3为0，因为root 为空无法安装摄像头，如果递归下来需要为空的节点安装摄像头，显然不成立，该方案不正确。而如果要求不用安装摄像头，不论是否需要被覆盖都是满足要求的，安装方案怎么都成立，且不消耗摄像头。

class Solution {
    public int minCameraCover(TreeNode root) {
        // 三个状态
        int[] res = minCam(root);
        return Math.min(res[0], res[2]);
    }

    public int[] minCam(TreeNode root) {
        if (root == null) {
            return new int[] {1000, 0, 0};
        }
        int[] left = minCam(root.left);
        int[] right = minCam(root.right);
        int[] ret = new int[3];
        ret[0] = Math.min(left[1] + right[1], Math.min(left[0] + right[1], left[1] + right[0])) + 1;
        ret[1] = Math.min(left[0] + right[0], left[0] + right[2]);
        ret[1] = Math.min(ret[1], Math.min(left[2] + right[0], left[2] + right[2]));
        ret[2] = Math.min(left[0] + right[0], Math.min(left[0] + right[2], left[2] + right[0]));
        return ret;
    }
}

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        Deque<TreeNode> st = new LinkedList<>();
        TreeNode cur = root;
        TreeNode pre = null;
        while (cur != null || !st.isEmpty()) {
            if (cur != null) {
                st.push(cur);
                cur = cur.left;
            } else {
                cur = st.peek();
                if (cur.right == null || cur.right == pre) {
                    pre = st.pop();
                    res.add(cur.val);
                    cur = null;
                } else {
                    cur = cur.right;
                }
            }
        }
        return res;
    }
}

class Solution {
    public void sortColors(int[] nums) {
        int i = 0;
        int lt = 0, gt = nums.length - 1;
        while (i <= gt) {
            if (nums[i] == 0) {
                swap(nums, i++, lt++);
            } else if (nums[i] == 2) {
                swap(nums, i, gt--);
            } else {
                i++;
            }
        }
    }

    public void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
