class Solution {
    public boolean isValid(String s) {
        HashMap<Character, Character> map = new HashMap<>();
        map.put('(', ')');
        map.put('[', ']');
        map.put('{', '}');
        Deque<Character> st = new LinkedList<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '(' || c == '[' || c == '{') {
                st.push(c);
            } else {
                if (st.isEmpty() || map.get(st.peek()) != c) {
                    return false;
                } else {
                    st.pop();
                }
            }
        }
        return st.isEmpty();
    }
}

class Solution {
    public boolean find132pattern(int[] nums) {
        int n = nums.length;
        Deque<Integer> d = new ArrayDeque<>();
        int k = Integer.MIN_VALUE;
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] < k) return true;
            while (!d.isEmpty() && d.peekLast() < nums[i]) {
                // 事实上，k 的变化也具有单调性，直接使用 k = pollLast() 也是可以的
                k = Math.max(k, d.pollLast()); 
            }
            d.addLast(nums[i]);
        }
        return false;
    }
}

class MyStack {

    private Queue<Integer> q1;
    private Queue<Integer> q2;

    public MyStack() {
        q1 = new ArrayDeque<>();
        q2 = new ArrayDeque<>();
    }
    
    public void push(int x) {
        q1.add(x);
    }
    
    public int pop() {
        while (q1.size() > 1) {
            q2.add(q1.poll());
        }
        int val = q1.poll();
        Queue<Integer> temp = q1;
        q1 = q2;
        q2 = temp;
        return val;
    }
    
    public int top() {
        int val = pop();
        q1.add(val);
        return val;
    }
    
    public boolean empty() {
        return q1.isEmpty() && q2.isEmpty();
    }
}

// 一个队列
class MyStack {

    private Queue<Integer> q;
    private int topElem;

    public MyStack() {
        q = new ArrayDeque<>();

    }
    
    public void push(int x) {
        q.add(x);
        topElem = x;
    }
    
    public int pop() {
        int size = q.size();
        while (size > 1) {
            if (size == 2)
                topElem = q.peek();
            q.add(q.poll());
            size--;
        }
        return q.poll();
    }
    
    public int top() {
        return topElem;
    }
    
    public boolean empty() {
        return q.isEmpty();
    }
}

class Solution {
    public int evalRPN(String[] tokens) {
        Deque<String> st = new LinkedList<>();
        st.push(tokens[0]);
        int i = 1;
        while (i < tokens.length) {
            String cur = tokens[i];
            if (cur.equals("+") || cur.equals("-") || cur.equals("*") || cur.equals("/")) {
                int op1 = Integer.parseInt(st.pop());
                int op2 = Integer.parseInt(st.pop());
                if (cur.equals("+"))
                    st.push(String.valueOf(op1 + op2));
                else if (cur.equals("-")) 
                    st.push(String.valueOf(op2 - op1));
                else if (cur.equals("*")) 
                    st.push(String.valueOf(op1 * op2));
                else
                    st.push(String.valueOf(op2 / op1));
            } else {
                st.push(cur);
            }
            i++;
        }
        return Integer.parseInt(st.peek());
    }
}

239. 滑动窗口最大值

对于一个固定窗口，一般都使用单调队列结构。因为窗口滑动的左右指针间距是固定的，不适合窗口扩大收缩得做法。

对于这个题目，双端队列保证单调不增。在滑动过程中，有窗口右边元素加入，这里需要保证队列得单调性；有窗口左边元素退出，如果退出元素在队列中，则需要删掉。

class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        Deque<Integer> q = new LinkedList<>();
        for (int i = 0; i < k; i++) {
            while (!q.isEmpty() && nums[i] > q.peekLast()) {
                q.pollLast();
            }
            q.addLast(nums[i]);
        }
        int[] res = new int[nums.length - k + 1];
        res[0] = q.peekFirst();
        for (int i = 1; i < nums.length - k + 1; i++) {
            int right = nums[i + k - 1];
            int left = nums[i-1];
            if (left == q.peekFirst())
                q.pollFirst();
            while (!q.isEmpty() && right > q.peekLast()) {
                q.pollLast();
            }
            q.addLast(right);
            res[i] = q.peekFirst();
        }
        return res;
    }
}
