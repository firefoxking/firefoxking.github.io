560. 和为 K 的子数组

第一个想法：利用前缀和来处理，首先得到nums数组的前缀和数组，然后两个for循环。这个解法时间复杂度O(n^2)，力扣上提交的耗时过长。

忘记了哈希表优化，难受

仔细一想确实是个好想法，理解也并不难。主要要把preSum - (preSum - k) == k这个转变过来。

  • 由于只关心次数，不关心具体的解，我们可以使用哈希表加速运算；

  • 由于保存了之前相同前缀和的个数，计算区间总数的时候不是一个一个地加，时间复杂度降到了 O(N)。

class Solution {
    public int subarraySum(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int sum = 0;
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            res += map.getOrDefault(sum - k, 0);
            map.put(sum, map.getOrDefault(sum, 0) + 1);
        }
        return res;
    }
}

class Solution {
    public int[][] generateMatrix(int n) {
        int[][] matrix = new int[n][n];
        int rowH = 0, rowL = n-1, colL = 0, colR = n-1;
        int value = 1;
        while (rowH <= rowL) {
            value = printMatrix(value, rowH, rowL, colL, colR, matrix);
            rowH++;
            rowL--;
            colL++;
            colR--;
        }
        return matrix;
    }

    public int printMatrix(int value, int rowH, int rowL, int colL, int colR, int[][] matrix) {
        for (int i = colL; i <= colR; i++) {
            matrix[rowH][i] = value++;
        }
        for (int i = rowH+1; i <= rowL; i++) {
            matrix[i][colR]  = value++;
        }
        for (int i = colR-1; i >= colL; i--) {
            matrix[rowL][i] = value++;
        }
        for (int i = rowL-1; i >= rowH+1; i--) {
            matrix[i][colL] = value++;
        }
        return value;
    }
}

class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int res = Integer.MAX_VALUE;
        int sum = 0, left = 0, right = 0;
        while (right < nums.length) {
            sum += nums[right];
            right++;
            while (sum >= target) {
                res = Math.min(res, right - left);
                sum -= nums[left];
                left++;
            }
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}

class Solution {
    public int lengthOfLongestSubstring(String s) {
        HashSet<Character> window = new HashSet<>();
        int left = 0, right = 0;
        int res = 0;
        while (right < s.length()) {
            char c = s.charAt(right);
            right++;
            while (window.contains(c)) {
                char d = s.charAt(left);
                left++;
                window.remove(d);
            }
            window.add(c);
            res = Math.max(res, right - left);
        }
        return res;
    }
}
// 方法二：用map继续优化一下
public class Solution {
    public int lengthOfLongestSubstring(String s) {
        int n = s.length(), ans = 0;
        Map<Character, Integer> map = new HashMap<>(); 
        for (int j = 0, i = 0; j < n; j++) {
            if (map.containsKey(s.charAt(j))) {
                i = Math.max(map.get(s.charAt(j)), i); 
            }
            ans = Math.max(ans, j - i + 1);
            map.put(s.charAt(j), j + 1);//下标 + 1 代表 i 要移动的下个位置
        }
        return ans;
    }
}

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            if (nums[i] > 0) break;
            if (i != 0 && nums[i] == nums[i-1])
                continue;
            int a = nums[i];
            int target = -a;
            int left = i + 1, right = nums.length - 1;
            while (left < right) {
                if (nums[left] + nums[right] == target) {
                    List<Integer> list = new ArrayList<>();
                    list.add(a);
                    list.add(nums[left]);
                    list.add(nums[right]);
                    res.add(list);
                    while (left < right && nums[left] == target - nums[right])
                        left++;
                } else if (nums[left] + nums[right] > target) {
                    right--;
                } else {
                    left++;
                }
            }
        }
        return res;
    }
}

class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        List<List<Integer>> res = new ArrayList<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (i != 0 && nums[i] == nums[i-1])
                continue;
            for (int j = i + 1; j < nums.length; j++) {
                if (j != i+1 && nums[j] == nums[j-1])
                    continue;
                int a = nums[i], b = nums[j];
                int left = j + 1, right = nums.length - 1;
                while (left < right) {
                    int sum = nums[left] + nums[right];
                    if (sum == target - a - b) {
                        List<Integer> list = new ArrayList<>();
                        list.add(a);
                        list.add(b);
                        list.add(nums[left]);
                        list.add(nums[right]);
                        res.add(list);
                        while (left < right && nums[left] == sum - nums[right])
                            left++;
                    } else if (sum > target - a - b) {
                        right--;
                    } else {
                        left++;
                    }
                }
            }
        }
        return res;
    }
}
