class Solution {
    public int rob(int[] nums) {
        int m = nums.length;
        if (m == 1)
            return nums[0];
        int[][] dp = new int[m+1][2];
        for (int i = 2; i <= m; i++) {
            dp[i][0] = Math.max(dp[i-1][0], dp[i-1][1]);
            dp[i][1] = dp[i-1][0] + nums[i-1];
        }
        int res = Math.max(dp[m][0], dp[m][1]);
        dp[1][1] = nums[0];
        for (int i = 2; i <= m; i++) {
            dp[i][0] = Math.max(dp[i-1][0], dp[i-1][1]);
            dp[i][1] = dp[i-1][0] + nums[i-1];
        }
        res = Math.max(res, dp[m][0]);
        return res;
    }
    
    public int rob(int[] nums) {
        int m = nums.length;
        if (m == 1)
            return nums[0];
        int[] dp = new int[2];
        for (int i = 1; i < m; i++) {
            int temp = dp[0];
            dp[0] = Math.max(dp[0], dp[1]);
            dp[1] = temp + nums[i];
        }
        int res = Math.max(dp[0], dp[1]);
        dp[1] = nums[0];
        dp[0] = 0;
        for (int i = 1; i < m; i++) {
            int temp = dp[0];
            dp[0] = Math.max(dp[0], dp[1]);
            dp[1] = temp + nums[i];
        }
        res = Math.max(res, dp[0]);
        return res;
    }
}

class Solution {

        return Math.max(dp[0], dp[1]);
    }

总结一下：我的做法与代码随想录、labuladong的都不一样，主要是状态定义这里。

  • 我的做法

    1

    dp[i..n][j..2]： dp[i][0] 表示不偷第i个房屋时能达到的最大数额， dp[i][1] 表示偷第i个房屋时能达到的最大数额。 思想与股票买卖类似。

  • 代码随想录

    1

    dp[i..n]: dp[i] 表示下标i（包括i）以内的房屋，最多可以偷窃的金额为dp[i]。这里比我的更快了一步，直接合并了i个房屋偷和不偷的值。

  • labuladong

    1

    dp[i..n]: dp[i] 表示从从第 i 间房子开始抢劫，最多能抢到的钱为 x， 与代码随想录是一样的。只是反过来了，遍历的时候也要反着遍历。

    class Solution {
    public int rob(TreeNode root) {
        int[] res = robTree(root);
        return Math.max(res[0], res[1]);
    }

    public int[] robTree(TreeNode root) {
        int[] res = new int[2];
        if (root == null)
            return res;
        int[] left = robTree(root.left);
        int[] right = robTree(root.right);
        res[0] = Math.max(left[0], left[1]) + Math.max(right[0], right[1]);
        res[1] = root.val + left[0] + right[0];
        return res;
    }
}

class Solution {
    public int[][] reconstructQueue(int[][] people) {
        int m = people.length;
        int[][] copy = new int[m][];
        for (int i = 0; i < m; i++) {
            copy[i] = new int[]{people[i][0], people[i][1]};
        }
        HashMap<int[], Integer> map = new HashMap<>();
        for (int i = 0; i < m; i++) {
            map.put(copy[i], i);
        }
        for (int i = 0; i < m; i++) {
            Arrays.sort(copy, i, m, (int[] a, int[] b) -> {
                if (a[1] == b[1])
                    return a[0] - b[0];
                return a[1] - b[1];
            });
            int num = copy[i][0];
            for (int j = i + 1; j < m; j++) {
                if (copy[j][0] <= num) {
                    copy[j][1] -= 1;
                }
            }
        }
        int[][] res = new int[m][];
        for (int i = 0; i < m; i++) {
            res[i] = people[map.get(copy[i])];
        }
        return res;

    }
}

class Solution {
    public int[][] reconstructQueue(int[][] people) {
        int m = people.length;
        int[][] copy = new int[m][];
        for (int i = 0; i < m; i++) {
            copy[i] = new int[]{people[i][0], people[i][1]};
        }
        HashMap<int[], Integer> map = new HashMap<>();
        int index = -1;
        for (int i = 0; i < m; i++) {
            map.put(copy[i], i);
            if (copy[i][1] == 0 && (index == -1 || copy[i][0] < copy[index][0]))
                index = i;
        }
        for (int i = 0; i < m; i++) {
            swap(copy, i, index);
            int num = copy[i][0];
            index = -1;
            for (int j = i + 1; j < m; j++) {
                if (copy[j][0] <= num) {
                    copy[j][1] -= 1;
                }
                if (copy[j][1] == 0 && (index == -1 || copy[j][0] < copy[index][0])) 
                    index = j;
            }
        }
        int[][] res = new int[m][];
        for (int i = 0; i < m; i++) {
            res[i] = people[map.get(copy[i])];
        }
        return res;

    }

    public void swap(int[][] arr, int i, int j) {
        int[] temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount+1];
        Arrays.fill(dp, amount+1);
        dp[0] = 0;
        for (int i = 0; i < coins.length; i++) {
            for (int j = coins[i]; j <= amount; j++) {
                dp[j] = Math.min(dp[j], dp[j-coins[i]] + 1);
            }
        }
        return dp[amount] == amount+1 ? -1 : dp[amount];
    }
}

class Solution {
    public int change(int amount, int[] coins) {
        int[] dp = new int[amount+1];
        dp[0] = 1;
        for (int i = 0; i < coins.length; i++) {
            for (int j = coins[i]; j <= amount; j++) {
                dp[j] += dp[j - coins[i]];
            }
        }
        return dp[amount];
    }
}
