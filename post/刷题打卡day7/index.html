class Solution {
    public int[] sortArray(int[] nums) {
        quickSort(nums, 0, nums.length - 1);
        return nums;
    }

    public void quickSort(int[] arr, int left, int right) {
        if (left >= right) {
            return;
        }
        int pIndex = partition(arr, left, right);
        quickSort(arr, left, pIndex - 1);
        quickSort(arr, pIndex + 1, right);
    }

    public int partition(int[] arr, int left, int right) {
        int pivot = arr[left];
        int lt = left;
        for (int i = left + 1; i <= right; i++) {
            if (arr[i] < pivot) {
                lt++;
                swap(arr, lt, i);
            }
        }
        swap(arr, left, lt);
        return lt;
    }

    public void swap(int[] arr, int i, int j) {
        if (i == j)
            return;
        arr[i] ^= arr[j];
        arr[j] ^= arr[i];
        arr[i] ^= arr[j];
    }
}

class Solution {

    int maxDepth = 0;
    public int maximumInvitations(int[] favorite) {
        int n = favorite.length;
        Map<Integer, List<Integer>> g = new HashMap<>();
        // 入度
        int[] deg = new int[n];
        for (int i = 0; i < n; i++) {
            int w = favorite[i];
            g.putIfAbsent(w, new ArrayList<>());
            g.get(w).add(i);
            deg[w]++;
        }
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < n; i++) {
            if (deg[i] == 0)
                q.add(i);
        }
        while (!q.isEmpty()) {
            int v = q.poll();
            if (--deg[favorite[v]] == 0)
                q.add(favorite[v]);
        }

        int ans = 0;
        int sum = 0;
        boolean[] vis = new boolean[n];
        for (int i = 0; i < n; i++) {
            if (!vis[i] && deg[i] > 0) {
                List<Integer> ring = new ArrayList<>();
                foundRing(ring, vis, favorite, i);
                int size = ring.size();
                if (size == 2) {
                    int v = ring.get(0);
                    int w = ring.get(1);
                    maxDepth = 0;
                    rdfs(v, w, g, 1);
                    sum += maxDepth;
                    maxDepth = 0;
                    rdfs(w, v, g, 1);
                    sum += maxDepth;
                    ans = Math.max(ans, sum);
                } else {
                    ans = Math.max(size, ans);
                }
            }
        }
        return ans;
    }

    public void foundRing(List<Integer> ring, boolean[] vis, int[] favorite, int i) {
        vis[i] = true;
        ring.add(i);
        if (vis[favorite[i]]) return;
        foundRing(ring, vis, favorite, favorite[i]);
    }
    public void rdfs(int cur, int last, Map<Integer, List<Integer>> rmap, int depth) {
        maxDepth = Math.max(maxDepth, depth);
        if (!rmap.containsKey(cur)) return;
        for (int i : rmap.get(cur)) {
            //不与上一个相等，如果与上上个相等，意味着构成大于2的环了，那么就不会走这里过
            if (i != last) { 
                rdfs(i, cur, rmap, depth + 1);
            }
        }
    }
}

class Solution {
    public String longestWord(String[] words) {
        Arrays.sort(words, (o1, o2) -> {
            if (o1.length() == o2.length()) {
                return o2.compareTo(o1);
            }
            return o1.length() - o2.length();
        });
        String res = "";
        HashSet<String> set = new HashSet<>();
        set.add(res);
        for (String word : words) {
            if (set.contains(word.substring(0, word.length() - 1))) {
                res = word;
                set.add(word);
            }
        }
        return res;
    }
}

class Solution {
    public boolean isInterleave(String s1, String s2, String s3) {
        if (s3.length() != s1.length() + s2.length())
            return false;
        char[] str1 = s1.toCharArray();
        char[] str2 = s2.toCharArray();
        char[] str3 = s3.toCharArray();
        return dfs(str1, str2, str3, 0, 0, 0);
    }

    public boolean dfs(char[] str1, char[] str2, char[] str3, int i, int j, int k) {
        if (k == str3.length)
            return true;
        char a = i >= str1.length ? ' ' : str1[i];
        char b = j >= str2.length ? ' ' : str2[j];
        char c = str3[k];
        boolean ret = false;
        if (c == a)
            ret |= dfs(str1, str2, str3, i+1, j, k+1);
        if (c == b)
            ret |= dfs(str1, str2, str3, i, j+1, k+1);
        return ret;
    }
}

用 递归其实是有很多重复运算的。

可以据此改为记忆化的DFS 解法，对前面代码略加改动增加一个备忘录memo即可，当然也由此可以改成动态规划。能用动态规划的一定能用递归，加入记忆化也很容易，有的题目动态规划好入手，有的不好入手，或者边界情况比较难确定。DFS+记忆化是一个不错的保底思路，时间复杂度和DP是一样的

class Solution {
    public boolean isInterleave(String s1, String s2, String s3) {
        int n = s1.length(), m = s2.length(), t = s3.length();

        if (n + m != t) {
            return false;
        }

        boolean[][] f = new boolean[n + 1][m + 1];

        f[0][0] = true;
        for (int i = 0; i <= n; ++i) {
            for (int j = 0; j <= m; ++j) {
                int p = i + j - 1;
                if (i > 0) {
                    f[i][j] = f[i][j] || (f[i - 1][j] && s1.charAt(i - 1) == s3.charAt(p));
                }
                if (j > 0) {
                    f[i][j] = f[i][j] || (f[i][j - 1] && s2.charAt(j - 1) == s3.charAt(p));
                }
            }
        }

        return f[n][m];
    }
}
