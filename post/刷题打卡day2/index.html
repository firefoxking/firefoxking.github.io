class Solution {
    public int trap(int[] height) {
        int n = height.length;
        int left = 0, right = n - 1;
        int res = 0;
        int leftMax = 0, rightMax = 0;
        while (left < right) {
            if (height[left] < height[right]) {
                leftMax = Math.max(leftMax, height[left]);
                res += leftMax - height[left];
                left++;
            } else {
                rightMax = Math.max(rightMax, height[right]);
                res += rightMax - height[right];
                right--;
            }
        }
        return res;
    }
}

class Solution {
    public int[] nextGreaterElements(int[] nums) {
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[index] < nums[i])
                index = i;
        }
        int n = nums.length;
        int[] res = new int[n];
        Deque<Integer> q = new LinkedList<>();
        res[index] = -1;
        Arrays.fill(res, -1);
        int i = index + 1;
        boolean flag = false;
        while (true) {
            if (i == index+1) {
                if (flag == false)
                    flag = true;
                else
                    break;
            }
            if (i == n) {
                i = 0;
            }

            while (!q.isEmpty() && nums[i] > nums[q.peek()]) {
                int temp = q.pop();
                res[temp] = nums[i];
            }
            q.push(i);
            i++;
        }
        return res;
    }
}

class Solution {
    public int[] nextGreaterElements(int[] nums) {
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[index] < nums[i])
                index = i;
        }
        int n = nums.length;
        int[] res = new int[n];
        Deque<Integer> q = new LinkedList<>();
        res[index] = -1;
        q.push(nums[index]);
        int i = index - 1;
        while (true) {
            if (i == -1)
                i = n - 1;
            if (i == index) break;
            while (!q.isEmpty() && nums[i] >= q.peek()) {
                q.pop();
            }
            res[i] = q.isEmpty() ? -1 : q.peek();
            q.push(nums[i]);
            i--;
        }
        return res;
    }
}

class Solution {
    int res = 0;
    public int findTargetSumWays(int[] nums, int target) {
        backtrack(nums, 0, target);
        return res;
    }

    public void backtrack(int[] nums, int index, int target) {
        if (index == nums.length) {
            if (target == 0)
                res++;
            return;
        }
        backtrack(nums, index+1, target - nums[index]);
        backtrack(nums, index+1, target + nums[index]);
    }
}

动态规划

把问题转变后就变成在数组中寻找一些数加起来刚好等于P， 典型的背包问题。状态定义用一维，用二维的先想想。

dp[i][j] 表示前i个数刚好凑到j的方案数
转移方程就是：dp[i][j] = dp[i-1][j] || dp[i][j] = dp[i-1][j-nums[i]] + dp[i-1][j]); 代表对每个数都可以选择凑和不凑，两者方案数累加，如果不满足凑的要求（也就是数比j还大，无法凑出负数可以不考虑，这是建立在非负整数的前提下的），就只有不凑一种选择。

dp[j] 表示刚好凑到数字和i的方案数
转移方程就是：dp[j] = sum(dp[j-nums[i]])

class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++)
            sum += nums[i];
        target += sum;
        if (target < 0 || target % 2 == 1)
            return 0;
        int P = target / 2;
        int[] dp = new int[P + 1];
        dp[0] = 1;
        for (int i = 0; i < nums.length; i++) {
            for (int j = P; j >= nums[i]; j--) {
                dp[j] += dp[j-nums[i]];
            }
        }
        return dp[P];
    }
    
    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++)
            sum += nums[i];
        target += sum;
        if (target < 0 || target % 2 == 1)
            return 0;
        int P = target / 2;
        int n = nums.length;
        int[][] dp = new int[n+1][P + 1];
        for (int i = 0; i < n; i++)
            dp[i][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= P; j++) {
                if (j < nums[i-1])
                    dp[i][j] = dp[i-1][j];
                else
                    dp[i][j] = dp[i-1][j] + dp[i-1][j-nums[i-1]];
            }
        }
        return dp[n][P];
    }
}

class Solution {
    public int findMaxForm(String[] strs, int m, int n) {
        int[][] dp = new int[m+1][n+1];
        for (int i = 0; i < strs.length; i++) {
            int zeros = countZero(strs[i]);
            for (int j = m; j >= zeros; j--) {
                int ones = countOne(strs[i]);
                for (int k = n; k >= ones; k--) {
                    dp[j][k] = Math.max(dp[j][k], dp[j-zeros][k-ones] + 1);
                }
            }
        }
        return dp[m][n];
    }

    public int countZero(String str) {
        int ret = 0;
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == '0')
                ret++;
        }
        return ret;
    }

    public int countOne(String str) {
        int ret = 0;
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == '1')
                ret++;
        }
        return ret;
    }
}
